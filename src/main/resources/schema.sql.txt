
CREATE TABLE Person (
                id_person INT AUTO_INCREMENT NOT NULL,
                Name VARCHAR(50) NOT NULL,
                Surname VARCHAR(50) NOT NULL,
                Email VARCHAR(50),
                Password VARCHAR(264) NOT NULL,
                PRIMARY KEY (id_person)
);


CREATE TABLE Transactions (
                id_transaction INT NOT NULL,
                Commentaire VARCHAR(256) NOT NULL,
                Amount DECIMAL(11,2) NOT NULL,
                Willpayperson INT NOT NULL,
                Willbepayedperson INT NOT NULL,
                PRIMARY KEY (id_transaction)
);


CREATE TABLE Bank_Account (
                id_account INT AUTO_INCREMENT NOT NULL,
                Amount DOUBLE PRECISION NOT NULL,
                Accountholder INT NOT NULL,
                PRIMARY KEY (id_account)
);


ALTER TABLE Transactions ADD CONSTRAINT person_transactions_fk
FOREIGN KEY (Willpayperson)
REFERENCES Person (id_person)
ON DELETE NO ACTION
ON UPDATE NO ACTION;

ALTER TABLE Transactions ADD CONSTRAINT person_transactions_fk1

REFERENCES Person (id_person)

ALTER TABLE Bank_Account ADD CONSTRAINT person_bank_account_fk
